# Database Configuration untuk Prisma + Supabase
# Copy file ini dan rename menjadi .env.local
# Kemudian isi dengan credentials Supabase Anda

# Supabase Database URLs
# Dapatkan dari: Supabase Dashboard → Settings → Database → Connection String

# Connection pooling URL (untuk serverless functions)
# Port 6543 menggunakan PgBouncer untuk connection pooling
DATABASE_URL="postgresql://postgres:[YOUR-PASSWORD]@[YOUR-PROJECT-REF].supabase.co:6543/postgres?pgbouncer=true"

# Direct connection URL (untuk migrations)
# Port 5432 adalah direct connection ke PostgreSQL
DIRECT_URL="postgresql://postgres:[YOUR-PASSWORD]@[YOUR-PROJECT-REF].supabase.co:5432/postgres"

# Cara mendapatkan credentials:
# 1. Login ke https://app.supabase.com
# 2. Pilih project Anda (database: dashboard(TA))
# 3. Pergi ke Settings → Database
# 4. Scroll ke "Connection String" section
# 5. Pilih "URI" tab
# 6. Copy connection string
# 7. Ganti [YOUR-PASSWORD] dengan database password Anda
# 8. Ganti [YOUR-PROJECT-REF] dengan project reference Anda

# Contoh (JANGAN GUNAKAN INI, INI HANYA CONTOH):
